% Largeur de chaque gabarit pour tenir sur deux colonnes
\newcommand{\Largeur}{8.5}

% GABARIT MULTIPLES (Accolade en haut)
\newcommand{\gabaritMultiple}[3]{%
    \begin{tikzpicture}[scale=1]
        \useasboundingbox (0, -1) rectangle (\Largeur, 3.5);
        \pgfmathsetmacro{\w}{\Largeur/#1}
        
        \draw[thick, fill=#2!20] (0, 1.2) rectangle (\Largeur, 2.4);
        \draw[dotted, thick, line cap=round] ({\Largeur/2 - 1}, 1.8) -- ({\Largeur/2 + 1}, 1.8);
        
        \draw[decorate, decoration={brace, amplitude=8pt, raise=2pt}, thick]
            (0, 2.4) -- (\Largeur, 2.4) node[midway, above=10pt, font=\bfseries] {#3};
            
        \foreach \i in {1,...,#1} {
            \pgfmathsetmacro{\xpos}{(\i-1)*\w}
            \draw[thick, fill=#2!40] (\xpos, 0) rectangle +(\w, 1.2);
            \pgfmathsetmacro{\dotw}{min(\w*0.2, 0.4)}
            \draw[dotted, thick, line cap=round] (\xpos+\w/2-\dotw, 0.6) -- (\xpos+\w/2+\dotw, 0.6);
        }
    \end{tikzpicture}%
}

% GABARIT FRACTIONS (Accolade en bas)
\newcommand{\gabaritFraction}[3]{%
    \begin{tikzpicture}[scale=1]
        \useasboundingbox (0, -1) rectangle (\Largeur, 3.5);
        \pgfmathsetmacro{\w}{\Largeur/#1}
        
        \draw[thick, fill=#2!20] (0, 1.2) rectangle (\Largeur, 2.4);
        \draw[dotted, thick, line cap=round] ({\Largeur/2 - 1}, 1.8) -- ({\Largeur/2 + 1}, 1.8);
        
        \foreach \i in {1,...,#1} {
            \pgfmathsetmacro{\xpos}{(\i-1)*\w}
            \draw[thick, fill=#2!40] (\xpos, 0) rectangle +(\w, 1.2);
            \pgfmathsetmacro{\dotw}{min(\w*0.2, 0.4)}
            \draw[dotted, thick, line cap=round] (\xpos+\w/2-\dotw, 0.6) -- (\xpos+\w/2+\dotw, 0.6);
        }
        
        \draw[decorate, decoration={brace, amplitude=6pt, mirror, raise=2pt}, thick]
            (0, 0) -- (\w, 0) node[midway, below=8pt, font=\bfseries] {#3};
    \end{tikzpicture}%
}

% COMMANDE GÉNÉRALE
\newcommand{\barrePourcent}[2]{%
    \begin{tikzpicture}[scale=0.88]
        % Boîte invisible pour forcer un centrage parfait
        \useasboundingbox (0, -1.5) rectangle (\LargeurTotale, 4.5);
        
        \pgfmathsetmacro{\w}{\LargeurTotale/#1}
        \pgfmathsetmacro{\p}{int(100/#1)}
        
        % Bloc total
        \draw[thick, fill=color100!40] (0, 1.8) rectangle (\LargeurTotale, 3.6);
        \draw[dotted, thick, line cap=round] ({\LargeurTotale/2 - 2}, 2.4) -- ({\LargeurTotale/2 + 2}, 2.4);
        
        % Accolade 100%
        \draw[decorate, decoration={brace, amplitude=10pt, raise=2pt}, thick]
            (0, 3.6) -- (\LargeurTotale, 3.6)
            node[midway, above=12pt, font=\bfseries\Large] {100 \%};
        
        % Cases du bas
        \foreach \i in {1,...,#1} {
            \pgfmathsetmacro{\xpos}{(\i-1)*\w}
            
            \draw[thick, fill=#2!40] (\xpos, 0) rectangle +(\w, 1.8);
            
            \pgfmathsetmacro{\dotwidth}{\w * 0.22}
            \draw[dotted, thick, line cap=round]
                (\xpos + \w/2 - \dotwidth, 0.6) -- (\xpos + \w/2 + \dotwidth, 0.6);
            
            \draw[decorate, decoration={brace, amplitude=6pt, mirror, raise=2pt}, thick]
                (\xpos, 0) -- (\xpos + \w, 0)
                node[midway, below=10pt, font=\bfseries] {\p \%};
        }
    \end{tikzpicture}%
}

\newcommand{\barrePourcentUn}{%

        % La même boîte invisible pour aligner avec les autres
        \useasboundingbox (0, -1.5) rectangle (\LargeurTotale, 4.5);
        
        \pgfmathsetmacro{\w}{\LargeurTotale/100}
        
        % Bloc total
        \draw[thick, fill=color100!40] (0, 1.8) rectangle (\LargeurTotale, 3.6);
        \draw[dotted, thick, line cap=round] ({\LargeurTotale/2 - 2}, 2.4) -- ({\LargeurTotale/2 + 2}, 2.4);
        
        % Accolade 100%
        \draw[decorate, decoration={brace, amplitude=10pt, raise=2pt}, thick]
            (0, 3.6) -- (\LargeurTotale, 3.6)
            node[midway, above=12pt, font=\bfseries\Large] {100 \%};

        % Correction : un seul grand bloc pour fixer la largeur exacte
        \draw[thick, fill=color1!35] (0,0) rectangle (\LargeurTotale,1.8);
        
        % Et on trace uniquement les lignes de séparation intérieures
        \foreach \i in {1,...,99} {
            \pgfmathsetmacro{\xpos}{\i*\w}
            \draw[thick] (\xpos,0) -- (\xpos,1.8);
        }

        % Accolade sous la 1re case
        \draw[decorate, decoration={brace, amplitude=5pt, mirror, raise=2pt}, thick]
            (0,0) -- (\w,0)
            node[midway, below=9pt, font=\bfseries\small] {1 \%};

        % Flèche vers la 2e case
        \draw[-{Latex[length=1.8mm]}, thick]
            (1.8,-0.55) -- ({1.5*\w},0.03);

        % Texte à compléter
        \node[font=\bfseries\small, anchor=west] at (1.9,-0.55) {Une case = ...};
    \end{tikzpicture}%
}

% Largeur totale commune à tous les gabarits
\newcommand{\LargeurTotale}{18}

% ==========================================
% MACROS POUR LE RECTO (Gabarits à compléter)
% ==========================================
\newcommand{\barreRecto}[2]{%
    \begin{tikzpicture}[scale=0.78]
        % Boîte invisible identique pour un alignement recto/verso parfait
        \useasboundingbox (0, -1.5) rectangle (\LargeurTotale, 4.5);
        
        \pgfmathsetmacro{\w}{\LargeurTotale/#1}
        
        % Bloc total
        \draw[thick, fill=#2!20] (0, 1.8) rectangle (\LargeurTotale, 3.6);
        \draw[dotted, thick, line cap=round] ({\LargeurTotale/2 - 2}, 2.4) -- ({\LargeurTotale/2 + 2}, 2.4);
        
        % Cases du bas
        \foreach \i in {1,...,#1} {
            \pgfmathsetmacro{\xpos}{(\i-1)*\w}
            \draw[thick, fill=#2!40] (\xpos, 0) rectangle +(\w, 1.8);
            
            \pgfmathsetmacro{\dotwidth}{\w * 0.22}
            \draw[dotted, thick, line cap=round]
                (\xpos + \w/2 - \dotwidth, 0.6) -- (\xpos + \w/2 + \dotwidth, 0.6);
        }
    \end{tikzpicture}%
}

\newcommand{\barreRectoUn}{%
    \begin{tikzpicture}[scale=0.78]
        \useasboundingbox (0, -1.5) rectangle (\LargeurTotale, 4.5);
        \pgfmathsetmacro{\w}{\LargeurTotale/100}
        
        % Bloc total
        \draw[thick, fill=color1!20] (0, 1.8) rectangle (\LargeurTotale, 3.6);
        \draw[dotted, thick, line cap=round] ({\LargeurTotale/2 - 2}, 2.4) -- ({\LargeurTotale/2 + 2}, 2.4);
        
        % Bloc des parts (100 cases)
        \draw[thick, fill=color1!35] (0,0) rectangle (\LargeurTotale,1.8);
        \foreach \i in {1,...,99} {
            \pgfmathsetmacro{\xpos}{\i*\w}
            \draw[thick] (\xpos,0) -- (\xpos,1.8);
        }

        % Indicateur à compléter pour 1 case
        \draw[-{Latex[length=1.8mm]}, thick]
            (1.8,-0.55) -- ({\w/2},0);
        \node[font=\bfseries\small, anchor=west] at (1.9,-0.55) {1 case = ...};
    \end{tikzpicture}%
}

% ==========================================
% MACROS POUR LE VERSO (Pourcentages inclus)
% ==========================================
\newcommand{\barreVerso}[2]{%
    \begin{tikzpicture}[scale=0.78]
        \useasboundingbox (0, -1.5) rectangle (\LargeurTotale, 4.5);
        
        \pgfmathsetmacro{\w}{\LargeurTotale/#1}
        \pgfmathsetmacro{\p}{int(100/#1)}
        
        % Bloc total avec 100% à l'intérieur
        \draw[thick, fill=#2!20] (0, 1.8) rectangle (\LargeurTotale, 3.6);
        \node[font=\bfseries\Large] at ({\LargeurTotale/2}, 2.7) {100 \%};
        
        % Cases du bas avec pourcentages à l'intérieur
        \foreach \i in {1,...,#1} {
            \pgfmathsetmacro{\xpos}{(\i-1)*\w}
            \draw[thick, fill=#2!40] (\xpos, 0) rectangle +(\w, 1.8);
            \node[font=\bfseries] at (\xpos + \w/2, 0.9) {\p \%};
        }
    \end{tikzpicture}%
}

\newcommand{\barreVersoUn}{%
    \begin{tikzpicture}[scale=0.78]
        \useasboundingbox (0, -1.5) rectangle (\LargeurTotale, 4.5);
        \pgfmathsetmacro{\w}{\LargeurTotale/100}
        
        % Bloc total avec 100% à l'intérieur
        \draw[thick, fill=color1!20] (0, 1.8) rectangle (\LargeurTotale, 3.6);
        \node[font=\bfseries\Large] at ({\LargeurTotale/2}, 2.7) {100 \%};

        % Bloc des parts
        \draw[thick, fill=color1!35] (0,0) rectangle (\LargeurTotale,1.8);
        \foreach \i in {1,...,99} {
            \pgfmathsetmacro{\xpos}{\i*\w}
            \draw[thick] (\xpos,0) -- (\xpos,1.8);
        }

        % Indicateur 1% (trop petit pour écrire dans la case)
        \draw[-{Latex[length=1.8mm]}, thick]
            (1.8,-0.55) -- ({\w/2},0);
        \node[font=\bfseries\small, anchor=west] at (1.9,-0.55) {1 case = 1 \%};
    \end{tikzpicture}%
}

% ==========================================
% MACROS : GABARITS REMPLIS (COLONNE GAUCHE)
% ==========================================
\newcommand{\gabaritRempli}[4]{%
    \begin{tikzpicture}[scale=1]
        \useasboundingbox (0, -0.3) rectangle (\Largeur, 2.7);
        \pgfmathsetmacro{\w}{\Largeur/#1}
        
        % Bloc total (Haut)
        \draw[thick, fill=#2!20] (0, 1.2) rectangle (\Largeur, 2.4);
        \node[font=\bfseries] at ({\Largeur/2}, 1.8) {#3};
        
        % Bloc des parts (Bas)
        \foreach \i in {1,...,#1} {
            \pgfmathsetmacro{\xpos}{(\i-1)*\w}
            \draw[thick, fill=#2!40] (\xpos, 0) rectangle +(\w, 1.2);
            \node[font=\bfseries] at (\xpos+\w/2, 0.6) {#4};
        }
    \end{tikzpicture}%
}

% ==========================================
% MACROS : GABARITS VIDES (COLONNE DROITE)
% ==========================================
\newcommand{\gabaritVide}[2]{%
    \begin{tikzpicture}[scale=1]
        \useasboundingbox (0, -0.3) rectangle (\Largeur, 2.7);
        \pgfmathsetmacro{\w}{\Largeur/#1}
        
        % Bloc total (Haut) avec pointillés
        \draw[thick, fill=#2!20] (0, 1.2) rectangle (\Largeur, 2.4);
        \draw[dotted, thick, line cap=round] ({\Largeur/2 - 1.5}, 1.8) -- ({\Largeur/2 + 1.5}, 1.8);
        
        % Bloc des parts (Bas) avec pointillés
        \foreach \i in {1,...,#1} {
            \pgfmathsetmacro{\xpos}{(\i-1)*\w}
            \draw[thick, fill=#2!40] (\xpos, 0) rectangle +(\w, 1.2);
            \pgfmathsetmacro{\dotw}{min(\w*0.25, 0.5)}
            \draw[dotted, thick, line cap=round] (\xpos+\w/2-\dotw, 0.6) -- (\xpos+\w/2+\dotw, 0.6);
        }
    \end{tikzpicture}%
}
